  %%-- Draw the card back and outline

    rectangle (\legacycardwidth,\legacycardheight);

    \node[text width=(\legacycardwidth-\strippadding-\stripwidth-2*\textpadding)*1cm,

          (\strippadding+\stripwidth+\textpadding,\strippadding)
    {\legacytextfontsize%
